grammar gram;
SEMI : ';';
WHITESPACE : ( ' ' | '\t' | '\n' | '\r' )+ -> skip;
COMMENT : ('//' (~[\n])* | '/*' .*? '*/') -> skip;
WHILE : 'while';
LP : [(];
RP : [)];
IF : 'if';
ELSE : 'else';
RETURN : 'return';
LBR : [{];
RBR : [}];
MINUS : '-';
NUM : [0-9]+ ;
OR : 'or';
AND : 'and';
NOT : 'not';
RELOP : '>='|'<='|'>'|'<'|'!='|'==';
PLUS : [+];
MULOP : [*/%];
EQ: [=];
TYPE : ('int'|'string'|'double');
STRING_CONSTANT : '"' ( '\\"' | ~["] )* '"';
PRINT : 'print';
INPUT : 'input';
OPEN : 'open';
READ : 'read';
WRITE : 'write';
CLOSE : 'close';
NOW : 'now';
CMA : ',';
ID : ([A-Za-z_])([A-Za-z0-9_])*;

start : program;
program : var_decl_list braceblock;
var_decl_list : var_decl SEMI var_decl_list | ;
var_decl : TYPE ID;
stmts : stmt stmts | ;
stmt : cond | loop | return_stmt SEMI | assign SEMI | func_call SEMI;
func_call : builtin_func_call;
builtin_func_call : PRINT LP expr RP | INPUT LP RP | OPEN LP expr RP | READ LP expr RP | WRITE LP expr CMA expr RP | CLOSE LP expr RP | NOW LP RP;
assign : ID EQ expr;
cond : IF LP expr RP braceblock | IF LP expr RP braceblock ELSE braceblock;
braceblock : LBR stmts RBR;
expr : orexp;
orexp : orexp OR andexp | andexp;
andexp : andexp AND notexp | notexp;
notexp : NOT notexp | rel;
rel : sum RELOP sum | sum;
sum : sum PLUS term | sum MINUS term | term;
term : term MULOP neg | neg;
neg : MINUS neg | factor;
factor : NUM | LP expr RP | ID | STRING_CONSTANT | func_call;
return_stmt : RETURN expr;
loop : WHILE LP expr RP braceblock;
